[docs]def F_Algorithm(S, L_functions):
    '''
    An implementation of F-Algorithm (add ref)
    Pairwise Algorithm to compute temporal paths in Stream Graph
    :param S: A Stream Graph
    :param L_functions: Functions according to the path's type (supported :
        - foremost path
        - shortest foremost path
        - shortest path
        - fastest shortest path
        - fastest path
        - shortest fastest path)
    :return:
    '''
    # Initialisation
    F = {}
    for n,np in zip(S.nodes,S.node_presence):
        for t0,t1 in zip(np[::2],np[1::2]):
            s = (t0,t1,n)
            F[s] = L_functions['initialisation'](s)
    temporal_adjacency_list = defaultdict(set) # currently present links in the form of adjacency list
    nodes_to_udpate = set() # Reached nodes so far
    E = S.augmented_ordered_links()
    for i in E:
        c = i[0]
        if c == 1: # Arrival
            _, t0, t1, u, v = i
            l = (t0, t1, u, v)
            temporal_adjacency_list[u].add((t1, v))
            temporal_adjacency_list[v].add((t1, u))
            nodes_to_udpate.add(u)
            nodes_to_udpate.add(v)
            F_update(F, [l], temporal_adjacency_list, nodes_to_udpate, L_functions)
        else:
            _, t1, u, v = i
            temporal_adjacency_list[u].remove((t1, v))
            temporal_adjacency_list[v].remove((t1, u))
    return F

[docs]def bfs_update(L,source, batch_arrival, temporal_adjacency_list, L_functions):
    '''
    Proceeds to browse every links present at instant :math:'t_0' in order to propagate the update on current possible paths

    :param L:
    :param u:
    :param v:
    :param t0:
    :param t1:
    :param temporal_adjacency_list:
    :param L_functions:
    :return:
    '''
    b = batch_arrival[0][0]
    Q = deque()
    visited = set()
    for (_, e, u1, u2) in batch_arrival:
        if u2 in L and u1 != source:
            Q.append((u2, u1, e))
            visited.add((u2,u1,e))
        if u1 in L and u2 != source:
            Q.append((u1, u2, e))
            visited.add((u1,u2,e))
    if Q:
        while Q:
            pred, w, end_t = Q.popleft()
            updated = L_functions['update'](L, b, end_t, pred, w)
            if updated:
                # removed= L_functions['domination'](L, w)
                for t, k in temporal_adjacency_list[w]:
                    if (w,k,t) not in visited and k != source:
                        visited.add((w,k,t))
                        Q.append((w, k, t))

[docs]def FoP_update(L, t0, t1, u, v):
    '''
    We update 'L' according to the link (t0,t1,u,v) and FoP properties.

    :param L:
    :param t0:
    :param t1:
    :param u:
    :param v:
    :return:
    '''
    new_arrival = t0
    # If v is already in L, we've reached v at t <= t0.
    if v not in L:
        L[v].add(new_arrival)
        return True
    return False



[docs]def FoP_postprocess(L, source):
    '''
    Post process elements in 'L' to obtain time to reach 'ttr'

    :param L:
    :param source:
    :return:
    '''
    ttr = {}
    for k, v in L.items():
        n = k[2]
        if n in ttr:
            ttr[n] = min(ttr[n], v[0] - source[0])
        else:
            ttr[n] = v[0] - source[0]
    return ttr

[docs]def SFoP_update(L, t0, t1, u, v):
    au, du = L[u][-1]
    new_arrival = t0
    new_distance = du + 1
    if v in L:
        av, dv = L[v][-1]
        if (new_arrival, new_distance) not in L[v] and (new_arrival == av and new_distance < dv):  # Â Domination Step Here
            done = False
            while not done:
                if new_arrival == av:
                    L[v].remove((av, dv))
                    if L[v]:
                        av, dv = L[v][-1]
                    else:
                        done = True
                else:
                    done = True
            L[v].add((new_arrival, new_distance))
        else:
            # We did not update L : don't propagate the path !
            return False
    else:
        L[v].add((new_arrival, new_distance))
    return True

    ttr = {}
    lengths = {}
    for k, v in L.items():

        if n in lengths:

[docs]def SFP_update(L, t0, t1, u, v):
    '''
    Update Lv from (t0,t1,u,v) according to SFP constraints.

    :param L:
    :param t0:
    :param t1:
    :param u:
    :param v:
    :return:
    '''
    au, su, du = L[u][-1]
    new_arrival = t0
    new_start = min(t1, su)
    new_distance = du + 1
    if v in L:
        av, sv, dv = L[v][-1]
        if (new_arrival, new_start, new_distance) not in L[v] and\
                (new_start > sv or (new_start == sv and new_distance < dv)
                 or (new_arrival<=new_start and new_distance<dv)):
            done = False
            while not done:
                if av == new_arrival:
                    L[v].remove((av, sv, dv))
                    if L[v]:
                        av, sv, dv = L[v][-1]
                    else:
                        done = True
                else:
                    done = True
            L[v].add((new_arrival, new_start, new_distance))
        else:
            return False
    else:
        L[v].add((new_arrival, new_start, new_distance))
    return True





[docs]def SFP_postprocess(L):
    latencies = {}
    length = {}
    for k, v in L.items():
        n = k[2]

        duration = min([i[0] - i[1] for i in v])
        if duration < 0:
            duration = 0
        min_length = min([i[2] for i in v if i[0] - i[1] == duration or i[1] > i[0]]) # Or condition for null latencies
        if n in latencies:
            latencies[n] = min(latencies[n], duration)
            length[n] = min(length[n], min_length)
        else:
            latencies[n] = duration
            length[n] = min_length
    return latencies, length
