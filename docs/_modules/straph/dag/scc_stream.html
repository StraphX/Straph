[docs]def merge_scc_streams(l_scc):
    '''
    Merge the every other stream into the first one, we assume there's no connection between them
    and links aren't constructed yet
    :param l_scc
    :return:
    '''
    scc1 = l_scc[0]
    if scc1.times:
        min_t, max_t = scc1.times[0], scc1.times[1]
    for scc in l_scc[1:]:
        scc1.c_nodes += scc.c_nodes
        if scc.times:
            min_t = min(min_t, scc.times[0])
            max_t = max(max_t, scc.times[1])
    if scc1.times:
        scc1.times = [min_t, max_t]
    return scc1

    '''
    Retrieve the connected component link_presence from a msgpack file
    :param input: Readable bytes file open(...,'rb')
    :return:
    '''
    with open(storage_path + str(times[0]) + "_" + str(times[1]) + '_scc_stream.pck', 'rb') as input:

        S = decode_scc_stream(R)
    return S

    def __init__(self,
                 times=None,
                 c_nodes=None,
                 c_links=None,
                 ):
        '''
        A basic constructor for a scc stream
        :param c_nodes : A list of SCC nodes (each c node represent a SCC : a set of nodes, a begin time, an end time)
        :param c_links : A list of directed link (each link represent two connected SCC)
        '''
        self.times = times
        if not c_nodes:
            self.c_nodes = []
        if not c_links:
            self.c_links = []

    def __del__(self):
        for n in self.c_nodes:
            del n
        del self.times, self.c_links

[docs]    def compute_links(self):
        # IF there's is some links, discard them.
        if self.c_links:
            self.c_links = []
        # Dict c_node to id
        dict_cnodes_2_id = {self.c_nodes[i]: i for i in range(self.size())}
        # c_Link : (c_node_id1,c_node_id2)
        dict_begin_time_to_cnodes = defaultdict(list)
        dict_end_time_to_cnodes = defaultdict(list)
        for cn in self.c_nodes:
            dict_begin_time_to_cnodes[cn.times[0]].append(cn)
            dict_end_time_to_cnodes[cn.times[1]].append(cn)

        # For each time_end_find a time_begin (if it exists) then compute the intersection of ndoes
        for te in dict_end_time_to_cnodes:
            if te in dict_begin_time_to_cnodes:
                # Double for loop in order to match pairwise
                for cn_parent in dict_end_time_to_cnodes[te]:
                    for cn_child in dict_begin_time_to_cnodes[te]:
                        if not cn_child.nodes.isdisjoint(cn_parent.nodes):
                            self.c_links.append((dict_cnodes_2_id[cn_parent], dict_cnodes_2_id[cn_child]))



[docs]    def fastest_path(self, source, destination, begin_time, refactor=False):
        i = 0
        start_comp = None
        # Get starting component
        for cn in self.c_nodes:
            if cn.times[0] <= begin_time <= cn.times[1]:
                if source in cn.nodes:
                    start_comp = i
                    break
            i += 1
        if not start_comp:
            raise ValueError("Node : "
                             + str(source) + " does not exists in the Stream Graph at time " + str(begin_time) + " !")
        # Iterate on link
        # Construct adjacency list (directed)
        a_l = defaultdict(list)
        for l in self.c_links:
            # IF the destination is accesible from begin time : OK
            if self.c_nodes[l[1]].times[0] >= begin_time:
                a_l[l[0]].append(l[1])
        #  Iterate
        max_time = max(cn.times[1] for cn in self.c_nodes) + 1

        #####################################
        # 1. Tester si la destination est dans le voisinage.
        # 2. Trier par temps de fin.
        #####################################
        # Custom BFS on DAG
        def bfs_scc(a_l, start_comp, destination, max_time):
            path_queue = [(start_comp, [start_comp])]
            fastest_arrival = max_time  # This variable, once assigned, is used as a threshold (yeah)
            while path_queue:
                (v, path) = path_queue.pop(0)
                if v in a_l and a_l[v]:
                    for w in a_l[v]:
                        comp = self.c_nodes[w]
                        if destination in comp.nodes:
                            fastest_arrival = min(fastest_arrival, comp.times[0])
                            yield path + [w]
                        elif comp.times[1] <= fastest_arrival:
                            path_queue.append((w, path + [w]))

        fastest_paths = list(bfs_scc(a_l, start_comp, destination, max_time))
        if not fastest_paths:
            raise ValueError("There is no path between"
                             + str(source) + " and " + str(destination) + "starting at " + str(begin_time) + " !")
        print("Fastest paths :", fastest_paths)
        path_times = [[self.c_nodes[i].times[0] for i in p] for p in fastest_paths]
        print("Path times :", path_times)
        min_t = min([p[-1] for p in path_times])
        print("Fastest time : ", min_t)
        if refactor:
            l_P = []
            for p, t in zip(fastest_paths, path_times):
                if t[-1] == min_t:
                    l_P.append(self.refactor_path(p, t, source, destination, begin_time))
            return l_P
        fp_paths = []
        fp_times = []
        for p, t in zip(fastest_paths, path_times):
            if t[-1] == min_t:
                fp_paths.append(p)
                fp_times.append(t)
        return fp_paths, fp_times

[docs]    def plot(self, colors=None, saving_path=None, format='pdf'):
        fig, ax = plt.subplots()
        rectangles = []
        labels = []
        unique_nodes = set()
        min_t = 0
        max_t = 0
        icol = 0
        if not colors:
            colors = get_cmap(len(self.c_nodes))
        for cn in self.c_nodes:
            if len(cn.nodes) != 1:
                rectangles.append(mpatch.Rectangle((cn.times[0], min(cn.nodes)),
                                                   width=cn.times[1] - cn.times[0],
                                                   height=max(cn.nodes) - min(cn.nodes) + 0.3,
                                                   edgecolor='k',
                                                   facecolor=colors(icol),
                                                   alpha=0.3
                                                   ))
            else:
                rectangles.append(mpatch.Rectangle((cn.times[0], min(cn.nodes)),
                                                   width=cn.times[1] - cn.times[0],
                                                   height=0.3,
                                                   edgecolor='k',
                                                   facecolor=colors(icol),
                                                   alpha=0.5
                                                   ))
            icol += 1
            labels.append("\n\n\n".join(reversed([str(n) for n in cn.nodes])))
            unique_nodes |= cn.nodes
            min_t = min(min_t, cn.times[0])
            max_t = max(max_t, cn.times[1])
        icol = 0
        for r, l in zip(rectangles, labels):
            eps = random.choice([1, -1]) * (random.random() / 10)
            ax.add_artist(r)
            rx, ry = r.get_xy()
            cx = rx + r.get_width() / 2.0 + eps
            cy = ry + r.get_height() / 2.0
            ax.annotate(str(icol), (cx, cy), color='k', weight='bold',
                        fontsize=8, ha='center', va='center')
            icol += 1
        icol = 0
        rectangles = []
        for cn in self.c_nodes:
            if len(cn.nodes) != 1:
                for n in cn.nodes:
                    rectangles.append(mpatch.Rectangle((cn.times[0], n),
                                                       width=cn.times[1] - cn.times[0],
                                                       height=0.3,
                                                       edgecolor='k',
                                                       facecolor=colors(icol),
                                                       alpha=0.5
                                                       ))
            icol += 1
        for r in rectangles:
            ax.add_artist(r)
        ax.set_xlim((min_t, max_t))
        ax.set_ylim((min(unique_nodes), max(unique_nodes) + 1))
        plt.yticks(range(len(unique_nodes)), range(len(unique_nodes)),
                   fontname='Ubuntu', fontsize=12, color='#666699')
        plt.xticks(range(int(min_t), int(max_t) + 1),
                   fontname='Ubuntu', fontsize=12, color='#476b6b')
        plt.ylabel("Nodes", fontname='Ubuntu', fontsize=12, color='#666699')
        plt.xlabel("t", fontname='Ubuntu', fontsize=12, color='#476b6b')
        plt.title("Condensation Graph of Stream Graph", fontname='Ubuntu', fontsize=14)
        for spine in plt.gca().spines.values():
            spine.set_visible(False)
        plt.tick_params(top=False, bottom=True, right=False, left=True, labelleft=True, labelbottom=True)
        if saving_path:

[docs]    def get_kcores(self, storage_path=None):
        L = defaultdict(list)

        def para_cores(comp, storage_path):
            return comp.get_kcores(storage_path)

        r = Parallel(n_jobs=8, mmap_mode='r+')(
            delayed(para_cores)(comp, storage_path) for comp in self.c_nodes if comp.size() > 1)
        for l in r:
            for k, v in l.items():
                L[k] += v
        return L
